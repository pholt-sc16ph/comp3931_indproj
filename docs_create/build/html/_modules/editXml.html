    textToSearch = "in_"
    textToReplace = "in"
    fileToSearch  = "Connectivity.txt"

    tempFile = open(fileToSearch, 'r+')

    for line in fileinput.input(fileToSearch):
        tempFile.write(line.replace(textToSearch, textToReplace))
    tempFile.close()
